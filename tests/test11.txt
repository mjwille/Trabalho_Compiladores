arr1 = char[7] : 'r' 'o' 'd' 'r' 'i' 'g' 'o';
arr2 = char[7] : 'r' 'o' 'd' 'o' 'l' 'f' 'o';

i = int : 0;

	count = 0

	loop(i : 0, i<10, i + 1) {

};
