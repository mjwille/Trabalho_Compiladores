/* Programa que soma os n primeiros n√∫meros
 * usando a linguagem do trabalho.
 */

i = int : 0;
acc = int : 0;
result = int : 0;

sumNumbers(n = int) = int {

	loop(i : 1, i <= n,  1) {
		acc = acc + i
	}

	return acc
};

main() = int {
	result = sumNumbers(10)
	print "The sum is: ", result
	return 0
};
