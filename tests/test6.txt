/* Programa que verifica se um dado número
 * é primo usando a linguagem do trabalho.
 */

is_prime(n = int) = bool {
	loop(i : 2, i < n, 1) {
		q = n / i
		rest = n - (q * i)
		if(rest == 0) then
			return 0
	}
	return 1
};

};
